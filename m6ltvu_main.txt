<script>
    let currentMode = 'all';
    let selectedFile = null;
    const creds = ["6768705f3841774b4a536b74", "376d53394d4d55583534674f", "6d6b6b46486e493038543056386b585a"];
    const username = "Faizanand";
    const repo = "New-upload";
    const baseDomain = "https://uploadmycs.netlify.app/";

    function hexDecode(h) {
        let s = '';
        for (let i = 0; i < h.length; i += 2) s += String.fromCharCode(parseInt(h.substr(i, 2), 16));
        return s;
    }

    function setMode(m) {
        currentMode = m;
        document.querySelectorAll('.mode-btn').forEach(b => b.classList.remove('active'));
        document.getElementById('btn-' + m).classList.add('active');
        selectedFile = null; 
        document.getElementById('display-text').innerHTML = 'REQUESTING DATA SOURCE<span class="file-name">CLICK TO INITIALIZE</span>';
        document.getElementById('drop-zone').classList.remove('file-selected');
        document.getElementById('exec-btn').disabled = true;
        document.getElementById('result-section').style.display = "none";
    }

    function triggerInput() { 
        document.getElementById('input-' + currentMode).click(); 
    }

    function handleFileSelect() {
        const input = document.getElementById('input-' + currentMode);
        selectedFile = input.files[0];
        if(selectedFile) {
            document.getElementById('display-text').innerHTML = 'SOURCE READY<span class="file-name">' + selectedFile.name + '</span>';
            document.getElementById('drop-zone').classList.add('file-selected');
            document.getElementById('exec-btn').disabled = false;
        }
    }

    async function executeUpload() {
        const token = hexDecode(creds[0]) + hexDecode(creds[1]) + hexDecode(creds[2]);
        const btn = document.getElementById('exec-btn');
        const bar = document.getElementById('bar');
        const status = document.getElementById('status-text');
        const percent = document.getElementById('percent-text');

        btn.disabled = true; 
        btn.innerText = "AUTHENTICATING...";
        document.getElementById('log-container').style.display = "block";

        const reader = new FileReader();
        reader.onload = async function() {
            const content = reader.result.split(",")[1];
            const fileName = Math.random().toString(36).substring(7) + "_" + selectedFile.name.replace(/\s+/g, '_');

            status.innerText = "UPLOADING_TO_SERVER";
            let p = 0;
            let interval = setInterval(() => {
                if(p < 98) { 
                    p += (p > 80 ? 0.5 : 2); 
                    bar.style.width = p + "%"; 
                    percent.innerText = Math.floor(p) + "%"; 
                }
            }, 50);

            try {
                const response = await fetch(`https://api.github.com/repos/${username}/${repo}/contents/${fileName}`, {
                    method: "PUT",
                    headers: { "Authorization": `token ${token}`, "Content-Type": "application/json" },
                    body: JSON.stringify({ message: "System Upload", content: content })
                });

                clearInterval(interval);

                if(response.ok) {
                    bar.style.width = "100%"; 
                    percent.innerText = "100%";
                    status.innerText = "GEN_MASKED_URL";
                    
                    const rawUrl = `https://raw.githubusercontent.com/${username}/${repo}/main/${fileName}`;
                    const maskedToken = btoa(rawUrl);
                    const finalOutput = `${baseDomain}?file=${maskedToken}`;
                    
                    document.getElementById('final-url').innerText = finalOutput;
                    document.getElementById('result-section').style.display = "block";
                    btn.innerText = "COMPLETED";
                } else { throw new Error(); }
            } catch (e) {
                alert("CRITICAL_ERROR: SYSTEM_FAILED");
                btn.disabled = false;
                btn.innerText = "RETRY_AUTH";
            }
        };
        reader.readAsDataURL(selectedFile);
    }

    function copyResult() {
        const text = document.getElementById('final-url').innerText;
        navigator.clipboard.writeText(text);
        const b = document.querySelector('.copy-btn');
        b.innerText = "DONE";
        setTimeout(() => b.innerText = "COPY", 2000);
    }

    // Logic to handle link clicks on the main site
    window.onload = function() {
        const params = new URLSearchParams(window.location.search);
        const file = params.get('file');
        if (file) {
            try { window.location.href = atob(file); } catch(e) {}
        }
    }
